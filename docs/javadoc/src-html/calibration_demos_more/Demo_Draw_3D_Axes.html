001package calibration_demos_more;

003import java.awt.Color;
004
005import ij.IJ;
006import ij.ImagePlus;
007import ij.ImageStack;
008import ij.plugin.PlugIn;
009import ij.process.ImageProcessor;
010import imagingbook.calibration.zhang.Camera;
011import imagingbook.calibration.zhang.ViewTransform;
012import imagingbook.calibration.zhang.testdata.ZhangData;
013import imagingbook.lib.ij.IjLogStream;
014import imagingbook.lib.settings.PrintPrecision;
015import imagingbook.lib.util.ResourceUtils;
016
017
018/**
019 * This plugin projects and draws the projected X/Y/Z coordinate axes
020 * for each of the given camera views.
021 * 
022 * @author W. Burger
023 * @version 2017-05-30
024 */
025public class Demo_Draw_3D_Axes implements PlugIn {
026
027        static Class<?> resourceRootClass = ZhangData.class;
028        static String resourceDir = "resources/";
029        static String resourceName = "CalibImageStack.tif";
030
031        static Color BackGroundColor = Color.white;
032        static Color LineColor = Color.magenta;
033        static Color xColor = Color.red;
034        static Color yColor = Color.green;
035        static Color zColor = Color.blue;
036
037        static double axisLength = 5;   
038        static boolean BeVerbose = false;
039
040        static {
041                IjLogStream.redirectSystem();
042                PrintPrecision.set(6);
043        }
044
045        @Override
046        public void run(String arg0) {
047                // create a 3D model:
048
049                double[] p0 = {0.0, 0.0, 0.0}; // 3D origin
050                double[] p1 = {axisLength, 0.0, 0.0};
051                double[] p2 = {0.0, axisLength, 0.0};
052                double[] p3 = {0.0, 0.0, axisLength};
053
054                // open the test image (stack):
055                ImagePlus testIm = ResourceUtils.openImageFromResource(resourceRootClass, resourceDir, resourceName);
056                if (testIm == null) {
057                        IJ.error("Could not open calibration images!");
058                        return;
059                }
060                testIm.show();
061
062                // get the camera intrinsics (typically by calibration):
063                Camera camera = ZhangData.getCameraIntrinsics();
064
065                // get all view parameters (extrinsics):
066                ViewTransform[] views = ZhangData.getAllViewTransforms();
067                int M = views.length;
068
069                // project and draw the model into the views:
070                ImageStack stack = testIm.getStack();
071                for (int i = 0; i < M; i++) {
072                        ImageProcessor ip = stack.getProcessor(i + 1);
073                        ip.setColor(xColor);
074                        drawProjectedSegment(ip, camera, views[i], p0, p1);
075                        ip.setColor(yColor);
076                        drawProjectedSegment(ip, camera, views[i], p0, p2);
077                        ip.setColor(zColor);
078                        drawProjectedSegment(ip, camera, views[i], p0, p3);
079                }
080                testIm.updateAndDraw();
081        }
082
083        private void drawProjectedSegment(ImageProcessor ip, Camera cam, ViewTransform V, double[] P1, double[] P2) {
084                double[] u1 = cam.project(V, P1);
085                int u1x = (int) Math.round(u1[0]);
086                int u1y = (int) Math.round(u1[1]);
087                double[] u2 = cam.project(V, P2);
088                int u2x = (int) Math.round(u2[0]);
089                int u2y = (int) Math.round(u2[1]);
090                ip.drawLine(u1x, u1y, u2x, u2y);
091        }
092
093
094}
