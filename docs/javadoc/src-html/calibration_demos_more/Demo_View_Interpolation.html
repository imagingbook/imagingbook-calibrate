024/**
025 * This plugin performs interpolation of views, given a sequence
026 * of key views.
027 * Translations (3D camera positions) are interpolated linearly.
028 * Pairs of rotations are interpolated by linear mixture
029 * of the corresponding quaternion representations (see
030 * http://www.opengl-tutorial.org/intermediate-tutorials/tutorial-17-quaternions/#How_do_I_interpolate_between_2_quaternions__).
031 *
032 * @author W. Burger
033 * @version 2017-05-30
034 *
035 */
036public class Demo_View_Interpolation implements PlugIn {
037
038        static Class<?> resourceRootClass = ZhangData.class;
039        static String resourceDir = "resources/";
040        static String resourceName = "CalibImageStack.tif";
041
042        static int NumberOfInterpolatedFrames = 10;
043        static double PeakHeightZ = -1.5;
044
045        static Color LineColor = Color.black;
046        static Color BackGroundColor = Color.white;
047        static boolean BeVerbose = false;
048
049        static {
050                IjLogStream.redirectSystem();
051                PrintPrecision.set(6);
052        }
053
054        @Override
055        public void run(String arg0) {
056                ImagePlus testIm = ResourceUtils.openImageFromResource(resourceRootClass, resourceDir, resourceName);
057
058                if (testIm == null) {
059                        IJ.error("Could not open calibration images!"); 
060                        return;
061                }
062
063                Camera cam = ZhangData.getCameraIntrinsics();
064                Point2D[] modelPoints = ZhangData.getModelPoints();
065
066                int w = testIm.getWidth();
067                int h = testIm.getHeight();
068                int M = testIm.getNSlices();
069
070                ImageStack animation = new ImageStack(w, h);
071
072                for (int a = 0; a < M; a++) {
073                        int b = (a + 1) % M;
074                        ViewTransform viewA = ZhangData.getViewTransform(a);    // view A
075                        ViewTransform viewB = ZhangData.getViewTransform(b);    // view B
076                        if (BeVerbose)
077                                System.out.println("**** doing views " + a +"/"+ ((a + 1) % M));
078
079                        Rotation rA = viewA.getRotation();
080                        Rotation rB = viewB.getRotation();
081                        double[] tA = viewA.getTranslation();
082                        double[] tB = viewB.getTranslation();
083
084                        // interpolation step k for view pair (a,b)
085                        for (int k = 0; k < NumberOfInterpolatedFrames; k++) {
086                                double alpha = (double) k / NumberOfInterpolatedFrames;
087
088                                // interpolate rotation
089                                Rotation rk = Lerp(rA, rB, alpha);
090
091                                // interpolate translation:
092                                double[] tk = Lerp(tA, tB, alpha);
093
094                                ViewTransform viewK = new ViewTransform(rk, tk);
095                                //viewK.print();
096
097                                // Create a new frame for the interpolated view
098                                // and project the target model:
099                                ImageProcessor frame = new ColorProcessor(w, h);
100                                if (BackGroundColor != null) {
101                                        frame.setColor(BackGroundColor);
102                                        frame.fill();
103                                }
104                                projectAndDrawPyramids(frame, cam, viewK, modelPoints);
105                                String title = String.format("frame-%d-%d", a, k);
106                                animation.addSlice(title, frame);
107                        }
108                }
109
110                new ImagePlus("Animation", animation).show();
111        }
112
113        Rotation Lerp(Rotation R0, Rotation R1, double alpha) {
114                Quaternion qa = MathUtil.toQuaternion(R0);
115                Quaternion qb = MathUtil.toQuaternion(R1);
116                return MathUtil.toRotation(Lerp(qa, qb, alpha));
117        }
118
119        /**
120         * Linear quaternion interpolation (LERP)
121         * @param Q0
122         * @param Q1
123         * @param a
124         * @return
125         */
126        Quaternion Lerp(Quaternion Q0, Quaternion Q1, double a) {
127                return Quaternion.add(Q0.multiply(1 - a), Q1.multiply(a));
128        }
129
130        /**
131         * Linear translation interpolation
132         * @param t0
133         * @param t1
134         * @param a
135         * @return
136         */
137        double[] Lerp(double[] t0, double[] t1, double a) {
138                double[] t01 = t0.clone();
139                for (int i = 0; i < t01.length; i++) {
140                        t01[i] = (1 - a) * t0[i] + a * t1[i];
141                }
142                return t01;
143        }
144
145        // ----------------------------------------------------------------------
146
147        // draw pyramids instead of squares at Z = PeakHeightZ (inch)
148        void projectAndDrawPyramids(ImageProcessor ip, Camera cam, ViewTransform view, Point2D[] modelPoints) {
149                GridPainter painter = new GridPainter(ip);
150                for (int i = 0; i < modelPoints.length; i += 4) {
151                        Point2D[] modelSq = new Point2D[4];
152                        Point2D[] imageSq = new Point2D[4];
153                        // 3D points p0,...,p3 define a model square in the Z=0 plane
154                        for (int j = 0; j < 4; j++) {
155                                modelSq[j] = modelPoints[i + j];
156                                imageSq[j] = MathUtil.toPoint2D(cam.project(view, modelSq[j]));
157                        }
158                        painter.lineCol = LineColor;
159                        painter.drawSquare(imageSq);
160
161                        // make the 3D pyramid peak:
162                        double[] modelPeak3d = new double[3];
163                        modelPeak3d[0] = (modelSq[0].getX() + modelSq[2].getX()) / 2;   // X
164                        modelPeak3d[1] = (modelSq[0].getY() + modelSq[2].getY()) / 2;   // Y
165                        modelPeak3d[2] = PeakHeightZ;   // Z
166
167                        // project and draw the pyramid peak:
168                        Point2D pk = MathUtil.toPoint2D(cam.project(view, modelPeak3d));
169                        painter.drawLine(imageSq[0], pk);
170                        painter.drawLine(imageSq[1], pk);
171                        painter.drawLine(imageSq[2], pk);
172                        painter.drawLine(imageSq[3], pk);
173                }
174        }
