001package calibration_demos_more;
002
003import ij.IJ;
004import ij.ImagePlus;
005import ij.ImageStack;
006import ij.plugin.PlugIn;
007import ij.process.ImageProcessor;
008import imagingbook.calibration.zhang.Camera;
009import imagingbook.calibration.zhang.RectificationMapping;
010import imagingbook.calibration.zhang.testdata.ZhangData;
011import imagingbook.lib.ij.IjLogStream;
012import imagingbook.lib.interpolation.InterpolationMethod;
013import imagingbook.lib.settings.PrintPrecision;
014import imagingbook.lib.util.ResourceUtils;
015import imagingbook.pub.geometry.mappings.Mapping;
016
017/**
018 * This plugin projects opens an image stack containing the 5 Zhang
019 * test images and removes the lens distortion based on the calibrated 
020 * camera parameters. The resulting (rectified) frames are shown
021 * in a new image stack.
022 * 
023 * @author W. Burger
024 * @version 2017-05-30
025 */
026public class Demo_Rectification implements PlugIn {
027
028        static boolean BeVerbose = false;
029
030        static Class<?> resourceRootClass = ZhangData.class;
031        static String resourceDir = "resources/";
032        static String resourceName = "CalibImageStack.tif";
033
034        static {
035                IjLogStream.redirectSystem();
036                PrintPrecision.set(6);
037        }
038
039        @Override
040        public void run(String arg0) {
041                // open the test image (stack):
042                ImagePlus testIm = ResourceUtils.openImageFromResource(resourceRootClass, resourceDir, resourceName);
043
044                if (testIm == null) {
045                        IJ.error("Could not open calibration images!");
046                        return;
047                }
048                testIm.show();
049                String title = testIm.getShortTitle();
050
051                // get the camera intrinsics (typically by calibration):
052                Camera camera = ZhangData.getCameraIntrinsics();
053
054                // create a special geometric mapping
055                Mapping mapping = new RectificationMapping(camera);
056
057                // rectify the images and create a new stack:
058                ImageStack distStack = testIm.getStack();
059                final int w = distStack.getWidth();
060                final int h = distStack.getHeight();
061                final int M = distStack.getSize();
062                ImageStack rectStack = new ImageStack(w, h);
063                for (int i = 0; i < M; i++) {
064                        IJ.showProgress(i, M);
065                        ImageProcessor source = distStack.getProcessor(i + 1);
066                        ImageProcessor target = source.createProcessor(w, h);
067                        mapping.applyTo(source, target, InterpolationMethod.Bicubic);
068                        rectStack.addSlice("frame"+ (i + 1), target);
069                }
070                new ImagePlus(title + "-rectified", rectStack).show();
071        }
072
073}
