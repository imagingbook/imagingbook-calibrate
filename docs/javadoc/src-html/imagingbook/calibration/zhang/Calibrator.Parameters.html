013/**
014 * This is an implementation of the camera calibration method 
015 * described in
016 *   Z. Zhang, "A flexible new technique for camera calibration",
017 *   IEEE Transactions on Pattern Analysis and Machine Intelligence, 
018 *   22(11), pp. 1330-1334, 2000.
019 * See also 
020 * <a href="http://research.microsoft.com/en-us/um/people/zhang/Calib/">
021 * http://research.microsoft.com/en-us/um/people/zhang/Calib/</a>
022 * and
023 * <a href="http://research.microsoft.com/en-us/um/people/zhang/Papers/TR98-71.pdf">
024 * http://research.microsoft.com/en-us/um/people/zhang/Papers/TR98-71.pdf</a>
025 * @author W. Burger
026 * @version 2018/12/27
027 */
028public class Calibrator {
029        
030        /**
031         * Inner class representing a set of parameters for instantiating
032         * new objects of type of {@link Calibrator}. 
033         * @author WB
034         */
035        public static class Parameters {
036                public boolean normalize = true;
037                public boolean assumeZeroSkew = false;
038                public int maxEvaluations = 1000;       // for LM-optimizer
039                public int maxIterations = 1000;        // for LM-optimizer
040                public int lensDistortionKoeffients = 2;
041                public boolean beVerbose = false;
042        }
043        
044        private int M;                                                          // the number of camera views
045        private final Point2D[] modelPts;                       // the sequence of 2D points in the planar model
046        private final List<Point2D[]> imgPntSet;        // list of vectors containing observed 2D image points for each view
047        
048        private Point2D[][] obsPts = null;
049        private final Parameters params;
050        private Camera initCam, finalCam;
051        private ViewTransform[] initViews, finalViews;
052        
053        // ------- constructors ------------------------------
054        
055        /**
056         * The only constructor. 
057         * @param params a parameter object (default parameters are used if {@code null} is passed)
058         * @param model a sequence of 2D points specifying the x/y coordinates of 
059         *                      the planar calibration pattern (assuming zero z-coordinates)
060         */
061        public Calibrator(Parameters params, Point2D[] model) {
062                this.params = (params != null) ? params : new Parameters();
063                this.modelPts = model;
064                this.imgPntSet = new ArrayList<Point2D[]>();
065        }
066        
067        /**
068         * Adds a new observation (a sequence of 2D image points) of the planar calibration pattern.
069         * @param pts a sequence of 2D image points
070         */
071        public void addView(Point2D[] pts) {
072                imgPntSet.add(pts);
073        }
074        
075        /**
076         * Performs the actual camera calibration based on the provided sequence of views.
077         * @return the estimated camera intrinsics as a {@link Camera} object
078         */
079        public Camera calibrate() {
080                M = imgPntSet.size();   // number of views to process
081                if (M < 2) {
082                        throw new IllegalStateException("Calibration: at least two views needed");
083                }
084                
085                obsPts = imgPntSet.toArray(new Point2D[0][]);
086                
087                // Step 1: Calculate the homographies for each of the given N views:
088                HomographyEstimator hest = new HomographyEstimator();
089                RealMatrix[] H_init = hest.estimateHomographies(modelPts, obsPts);
090                
091                // Step 2: Estimate the intrinsic parameters by linear optimization:
092                CameraIntrinsicsEstimator cis = new CameraIntrinsicsEstimator();
093                
094                RealMatrix A_init = cis.getCameraIntrinsics(H_init);
095                initCam = new Camera(A_init, new double[params.lensDistortionKoeffients]);
096                
097                // Step 3: calculate the extrinsic view parameters:
098                ExtrinsicViewEstimator eve = new ExtrinsicViewEstimator(A_init);
099                initViews = eve.getExtrinsics(H_init);
100                
101                // Step 4: Determine the lens distortion from initial estimates:
102                RadialDistortionEstimator rde = new RadialDistortionEstimator();
103                double[] distParams = rde.estimateLensDistortion(initCam, initViews, modelPts, obsPts);
104                Camera improvedCam = new Camera(A_init, distParams);
105                
106                // Step 5: Refine all parameters by non-linear optimization
107                NonlinearOptimizer optimizer = new NonlinearOptimizerAnalytic(modelPts, obsPts);
108//              NonlinearOptimizer optimizer = new NonlinearOptimizerNumeric(modelPts, obsPts);
109                optimizer.optimize(improvedCam, initViews);
110                finalCam = optimizer.getFinalCamera();
111                finalViews = optimizer.getFinalViews();
112
113                return finalCam;
114        }
115

119        @SuppressWarnings("unused")
120        private void printHomographies(RealMatrix[] homographies) {
121                int i = 0;
122                for (RealMatrix H : homographies) {
123                        i++;
124                        System.out.println("Homography " + i + ":");
125                        System.out.println(Matrix.toString(H.getData()));
126                }
127        }
128        
129        /**
130         * Calculates the squared projection error for a single view, associated
131         * with a set of observed image points.
132         * @param cam a camera model (camera intrinsics)
133         * @param view a view transformation (camera extrinsics)
134         * @param observed a set of observed image points
135         * @return the squared projection error (measured in pixel units)
136         */
137    public double getProjectionError(Camera cam, ViewTransform view, Point2D[] observed) {
138        double sqError = 0;
139                for (int j = 0; j < modelPts.length; j++) {
140                        double[] uv = cam.project(view, modelPts[j]);
141                        double[] UV = MathUtil.toArray(observed[j]);
142                        double du = uv[0] - UV[0];
143                        double dv = uv[1] - UV[1];
144                        sqError = sqError + du * du + dv * dv;
145                }
146        return sqError;
147    }
148    
149        /**
150         * Calculates the squared projection error for a sequence of views, associated
151         * with a sequence of observed image point sets.
152         * @param cam a camera model (camera intrinsics)
153         * @param views a sequence of view transformations (camera extrinsics)
154         * @param observed a sequence of sets of observed image points
155         * @return the squared projection error (measured in pixel units)
156         */
157    public double getProjectionError(Camera cam, ViewTransform[] views, Point2D[][] observed) {
158        double totalError = 0;
159        for (int i = 0; i < views.length; i++) {
160                totalError = totalError + getProjectionError(cam, views[i], observed[i]);
161        }
162        return totalError;
163    }

167    /**
168     * Returns the initial camera model (no lens distortion).
169     * @return the initial camera model
170     */
171    public Camera getInitialCamera() {
172        return initCam;
173    }
174    
175    /**
176     * Returns the final camera model (including lens distortion).
177     * @return the final camera model
178     */
179    public Camera getFinalCamera() {
180        return finalCam;
181    }
182    
183    /**
184     * Returns the sequence of initial camera views (extrinsics, no 
185     * lens distortion).
186     * @return the sequence of initial camera views
187     */
188    public ViewTransform[] getInitialViews() {
189        return initViews;
190    }
191    
192    /**
193     * Returns the sequence of final camera views (extrinsics, including 
194     * lens distortion).
195     * @return the sequence of final camera views
196     */
197    public ViewTransform[] getFinalViews() {
198        return finalViews;
199    }
