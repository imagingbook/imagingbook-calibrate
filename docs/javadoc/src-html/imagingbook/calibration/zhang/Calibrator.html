012/**
013 * This is an implementation of the camera calibration method 
014 * described in
015 *   Z. Zhang, "A flexible new technique for camera calibration",
016 *   IEEE Transactions on Pattern Analysis and Machine Intelligence, 
017 *   22(11), pp. 1330-1334, 2000.
018 * See also http://research.microsoft.com/en-us/um/people/zhang/Calib/ and
019 * http://research.microsoft.com/en-us/um/people/zhang/Papers/TR98-71.pdf
020 * @author W. Burger
021 * @version 2016/05/14
022 *
023 */
024public class Calibrator {
025        
026        public static class Parameters {
027                public boolean normalize = true;
028                public boolean assumeZeroSkew = false;
029                public int maxEvaluations = 1000;       // for LM-optimizer
030                public int maxIterations = 1000;        // for LM-optimizer
031                public int lensDistortionKoeffients = 2;
032                public boolean beVerbose = false;
033        }
034        
035//      private double alpha, beta, gamma, cx, cy;      // camera intrinsics
036//      private RealMatrix A = null;                            // camera intrinsics
037        
038        private int M;                                                          // the number of camera views
039        private final Point2D[] modelPts;                       // the sequence of 2D points in the planar model
040        private final List<Point2D[]> imgPntSet;        // list of vectors containing observed 2D image points for each view
041        
042        private Point2D[][] obsPts = null;
043        private final Parameters params;
044        private Camera initCam, finalCam;
045        private ViewTransform[] initViews, finalViews;
046        
047        // ------- constructors ------------------------------
048        
049        public Calibrator(Parameters params, Point2D[] model) {
050                this.params = params;
051                this.modelPts = model;
052                this.imgPntSet = new ArrayList<Point2D[]>();
053        }
054        
055        // attach a new observation (set of image points)
056        public void addView(Point2D[] pts) {
057                imgPntSet.add(pts);
058        }
059        
060        /**
061         * The actual calibration is being done here.
062         * 
063         * @return the estimated camera parameters as a {@link Camera} object
064         */
065        public Camera calibrate() {
066                M = imgPntSet.size();   // number of views to process
067                if (M < 2) {
068                        throw new IllegalStateException("Calibration: at least two views needed");
069                }
070                
071                obsPts = imgPntSet.toArray(new Point2D[0][]);
072                
073                // Step 1: Calculate the homographies for each of the given N views:
074                HomographyEstimator hest = new HomographyEstimator();
075                RealMatrix[] H_init = hest.estimateHomographies(modelPts, obsPts);
076                
077                // Step 2: Estimate the intrinsic parameters by linear optimization:
078                CameraIntrinsicsEstimator cis = new CameraIntrinsicsEstimator();
079                
080                RealMatrix A_init = cis.getCameraIntrinsics(H_init);
081                initCam = new Camera(A_init, new double[params.lensDistortionKoeffients]);
082                
083                // Step 3: calculate the extrinsic view parameters:
084                ExtrinsicViewEstimator eve = new ExtrinsicViewEstimator(A_init);
085                initViews = eve.getExtrinsics(H_init);
086                
087                // Step 4: Determine the lens distortion from initial estimates:
088                RadialDistortionEstimator rde = new RadialDistortionEstimator();
089                double[] distParams = rde.estimateLensDistortion(initCam, initViews, modelPts, obsPts);
090                Camera improvedCam = new Camera(A_init, distParams);
091                
092                // Step 5: Refine all parameters by non-linear optimization
093                NonlinearOptimizer optimizer = new NonlinearOptimizerAnalytic(modelPts, obsPts);
094//              NonlinearOptimizer optimizer = new NonlinearOptimizerNumeric(modelPts, obsPts);
095                finalCam = optimizer.optimize(improvedCam, initViews);
096                finalViews = optimizer.getFinalViews();
097
098                return finalCam;
099        }

106                int i = 0;
107                for (RealMatrix H : homographies) {
108                        i++;
109                        MathUtil.print("Homography " + i, H); 
110                        
111                }
112        }

114    public double getProjectionError(Camera cam, ViewTransform view, Point2D[] observed) {
115        double sqError = 0;
116                for (int j = 0; j < modelPts.length; j++) {
117                        double[] uv = cam.project(view, modelPts[j]);
118                        double[] UV = MathUtil.toArray(observed[j]);
119                        double du = uv[0] - UV[0];
120                        double dv = uv[1] - UV[1];
121                        sqError = sqError + du * du + dv * dv;
122                }
123        return sqError;
124    }
125    
126    public double getProjectionError(Camera cam, ViewTransform[] views, Point2D[][] observed) {
127        double totalError = 0;
128        for (int i = 0; i < views.length; i++) {
129                totalError = totalError + getProjectionError(cam, views[i], observed[i]);
130        }
131        return totalError;
132    }
