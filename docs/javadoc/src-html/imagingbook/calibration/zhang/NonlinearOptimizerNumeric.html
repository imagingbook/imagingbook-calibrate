009/**
010 * Nonlinear optimizer based on the Levenberg-Marquart method, where the Jacobian matrix
011 * is calculated numerically (i.e., by estimating the first partial derivatives from
012 * finite differences).
013 * @author WB
014 */
015public class NonlinearOptimizerNumeric extends NonlinearOptimizer {
016        
017        protected NonlinearOptimizerNumeric(Point2D[] modelPts, Point2D[][] obsPts) {
018                super(modelPts, obsPts);
019        }
020        
021        @Override
022        MultivariateVectorFunction makeValueFun() {
023                return new ValueFun();
024        }
025
026        @Override
027        MultivariateMatrixFunction makeJacobianFun() {
028                return new JacobianFun();
029        }
030
031        private class JacobianFun implements MultivariateMatrixFunction {
032                // THIS VERSION only calculates single blocks of the Jacobian!
033            public double[][] value(double[] params) {
034                double[][] J = new double[2 * M * N][params.length];    // the Jacobian matrix (initialized to zeroes!)
035                double[] refValues = new double[2 * M * N];     // values obtained with undisturbed parameters 
036                
037                double[] s = Arrays.copyOfRange(params, 0, camParLength);
038                Camera camOrig = new Camera(s);
039                
040                // Step 0: calculate all 2MN reference output values (for undisturbed parameters)
041                
042                for (int row = 0, i = 0; i < M; i++) {  // for all views
043                        int start = camParLength + i * viewParLength;
044                                double[] w = Arrays.copyOfRange(params, start, start + viewParLength);
045                                ViewTransform view = new ViewTransform(w);
046                        for (int j = 0; j < N; j++) {   // for all model points: calculate reference values
047                                double[] uv = camOrig.project(view, modelPts[j]);
048                                refValues[row + 0] = uv[0];
049                                refValues[row + 1] = uv[1];
050                                row = row + 2;
051                        }                
052                }
053                
054                // Step 1: calculate the leftmost (green) block of J associated with camera intrinsics
055                
056                for (int k = 0; k < s.length; k++) {    // for all parameters in s
057                        int col = k;
058                        double sk = s[k];                       // keep original parameter value        
059                        double delta = estimateDelta(sk);
060                        s[k] = s[k] + delta;            // modify parameter s_k
061                        Camera camMod = new Camera(s);  // modified camera
062                        
063                        for (int row = 0, i = 0; i < M; i++) {  // for all views
064                                int start = camParLength + i * viewParLength;
065                                double[] w = Arrays.copyOfRange(params, start, start + viewParLength);
066                                ViewTransform view = new ViewTransform(w);
067                                for (int j = 0; j < N; j++) {   // for all model points: calculate disturbed value
068                                        Point2D Pj = modelPts[j];
069                                        double[] uvMod = camMod.project(view, Pj);
070                                        J[row + 0][col] = (uvMod[0] - refValues[row + 0]) / delta;   // dX
071                                        J[row + 1][col] = (uvMod[1] - refValues[row + 1]) / delta;   // dY
072                                        row = row + 2;
073                                }    
074                        }
075                        s[k] = sk;      // return parameter s_k to original
076                }
077                
078                // Step 2: calculate the diagonal blocks, one for each view
079                
080                for (int i = 0; i < M; i++) {   // for all views/blocks
081                        int start = camParLength + i * viewParLength;
082                        double[] w = Arrays.copyOfRange(params, start, start + viewParLength);
083                        int col = s.length + i * w.length;              // leftmost matrix column of block i
084                        for (int k = 0; k < w.length; k++) {    // for all parameters in w
085                                double wk = w[k];                                       // keep original parameter w_k
086                                double delta = estimateDelta(wk);
087                                w[k] =  w[k] + delta;                           // modify parameter w_k
088                                ViewTransform view = new ViewTransform(w);
089                                int row = 2 * i * N;
090                                for (int j = 0; j < N; j++) {           // for all model points: calculate disturbed value
091                                        Point2D Pj = modelPts[j];
092                                        double[] uvMod = camOrig.project(view, Pj);
093                                        J[row + 0][col + k] = (uvMod[0] - refValues[row + 0]) / delta;   // dX
094                                        J[row + 1][col + k] = (uvMod[1] - refValues[row + 1]) / delta;   // dY
095                                        row = row + 2;
096                                } 
097                                w[k] = wk; // w[k] - DELTA;             // return parameter w_k to original
098                                //col++;
099                        }
100                }
101                
102//                      long endtime = System.nanoTime();
103//                      System.out.println("time diff = " + (endtime - starttime) + " ns");
104//              System.out.println("Jacobian inverse condition number = " + MathUtil.inverseConditionNumber(J));
105                return J;
106            }
107                
108                 // THIS VERSION calculates all entries of the Jacobian (not used)!
109            @SuppressWarnings("unused")
110            @Deprecated
111                public double[][] value(double[] params, boolean dummy) {
112                long starttime = System.nanoTime();
113                //System.out.println("getJacobianMatrix - NUMERICAL");
114                // M = number of views, N = number of model points
115                double[][] J = new double[2 * M * N][params.length];    // the Jacobian matrix
116                double[] refValues = new double[2 * M * N];     // function values obtained with undisturbed parameters 
117                
118                double[] s = Arrays.copyOfRange(params, 0, camParLength);
119                Camera cam = new Camera(s);
120                
121                // Step 0: calculate all 2MN reference output values (for undisturbed parameters)
122                
123                for (int row = 0, i = 0; i < M; i++) {  // for all views
124                        int start = camParLength + i * viewParLength;
125                                double[] w = Arrays.copyOfRange(params, start, start + viewParLength);
126                                ViewTransform view = new ViewTransform(w);
127                        for (int j = 0; j < N; j++) {   // for all model points: calculate reference values
128                                double[] uv = cam.project(view, modelPts[j]);
129                                refValues[row + 0] = uv[0];
130                                refValues[row + 1] = uv[1];
131                                row = row + 2;
132                        }                
133                }
134                
135                // Step 1: calculate all entries of the Jacobian (brute!)
136                
137                for (int k = 0; k < params.length; k++) {       // for ALL parameters
138                        int col = k;
139                        // calculate step width and modify the parameter vector c:
140                        double pk = params[k];
141                        double delta = estimateDelta(pk);
142                        params[k] = params[k] + delta;          // modify parameter c_k
143                        
144                        double[] smod = Arrays.copyOfRange(params, 0, camParLength);
145                        Camera camMod = new Camera(smod);       // modified camera
146                        
147                        for (int row = 0, i = 0; i < M; i++) {  // for all views
148                                int start = camParLength + i * viewParLength;
149                                double[] w = Arrays.copyOfRange(params, start, start + viewParLength);
150                                ViewTransform view = new ViewTransform(w);
151                                for (int j = 0; j < N; j++) {   // for all model points: calculate disturbed value
152                                        Point2D Pj = modelPts[j];
153                                        double[] uvMod = camMod.project(view, Pj);
154                                        J[row + 0][col] = (uvMod[0] - refValues[row + 0]) / delta;   // du
155                                        J[row + 1][col] = (uvMod[1] - refValues[row + 1]) / delta;   // dv      
156                                        row = row + 2;
157                                }    
158                        }
159                        params[k] = pk;
160                }
161               
162                long endtime = System.nanoTime();
163                        System.out.println("time diff = " + (endtime - starttime) + " ns");
164                return J;
165            }
166
167        }
168        
169        /**
170         * Returns a positive delta value adapted to the magnitude of the parameter x
171         * @param x
172         * @return
173         */
174    private double estimateDelta(double x) {
175        final double eps = 1.5e-8;      // = sqrt(2.2 * 10^{-16})
176        double dx = eps * Math.max(Math.abs(x), 1); 
177        // avoid numerical truncation problems (add and subtract again) - 
178        // not sure if this survives the compiler !?
179        double tmp = x + dx;
180        return tmp - x;
181    }

183}
