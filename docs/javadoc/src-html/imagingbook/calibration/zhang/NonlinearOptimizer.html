001package imagingbook.calibration.zhang;
002
003import java.awt.geom.Point2D;
004import java.util.Arrays;
005
006import org.apache.commons.math3.analysis.MultivariateMatrixFunction;
007import org.apache.commons.math3.analysis.MultivariateVectorFunction;
008import org.apache.commons.math3.fitting.leastsquares.LeastSquaresFactory;
009import org.apache.commons.math3.fitting.leastsquares.LeastSquaresOptimizer.Optimum;
010import org.apache.commons.math3.fitting.leastsquares.LevenbergMarquardtOptimizer;
011import org.apache.commons.math3.fitting.leastsquares.MultivariateJacobianFunction;
012import org.apache.commons.math3.linear.ArrayRealVector;
013import org.apache.commons.math3.linear.RealVector;
014
015/**
016 * Abstract super-class for non-linear optimizers used for
017 * final, overall optimization of calibration parameters.
018 * The actual optimization is performed by the sub-classes.
019 * @author WB
020 */
021public abstract class NonlinearOptimizer {
022
023        static int maxEvaluations = 1000;
024        static int maxIterations = 1000;
025        static boolean beVerbose = false;
026
027        final Point2D[] modelPts;
028        final Point2D[][] obsPts;
029        final int M;    // number of views
030        final int N;    // number of model points
031
032        private Camera initCam, finalCamera = null;
033        private ViewTransform[] initViews, finalViews = null;
034        int camParLength = 7;
035        int viewParLength = 6;
036
037        protected NonlinearOptimizer(Point2D[] modelPts, Point2D[][] obsPts) {
038                this.modelPts = modelPts;
039                this.obsPts = obsPts;
040                this.M = obsPts.length;
041                this.N = modelPts.length;
042        }
043
044        /**
045         * Performs Levenberg-Marquardt non-linear optimization to get better
046         * estimates of the parameters.
047         * @param cam the initial camera parameters
048         * @param views the initial view transforms
049         */
050        protected void optimize(Camera cam, ViewTransform[] views) {
051                this.initCam = cam;
052                this.initViews = views;
053                this.camParLength = cam.getParameterVector().length;
054                this.viewParLength = views[0].getParameters().length;
055
056                MultivariateVectorFunction V = makeValueFun();
057                MultivariateMatrixFunction J = makeJacobianFun();
058
059                RealVector start = makeInitialParameters(initCam, initViews);
060                RealVector observed = makeObservedVector();
061                
062                MultivariateJacobianFunction model = LeastSquaresFactory.model(V, J);
063                LevenbergMarquardtOptimizer lm = new LevenbergMarquardtOptimizer();
064                Optimum result = lm.optimize(LeastSquaresFactory.create(
065                                model,
066                                observed, 
067                                start, 
068                                null, 
069                                maxEvaluations, 
070                                maxIterations));
071
072                updateEstimates(result.getPoint());
073        }
074
075        abstract MultivariateVectorFunction makeValueFun();
076        abstract MultivariateMatrixFunction makeJacobianFun();
077        
078        
079        /**
080         * Common value function for optimizers defined in sub-classes.
081         */
082        protected class ValueFun implements MultivariateVectorFunction {
083                public double[] value(double[] params) {
084                        final double[] a = Arrays.copyOfRange(params, 0, camParLength);
085                        final Camera cam = new Camera(a);
086                        final double[] Y = new double[2 * M * N];
087                        int l = 0; 
088                        for (int i = 0; i < M; i++) {
089                                int m = camParLength + i * viewParLength;
090                                double[] w = Arrays.copyOfRange(params, m, m + viewParLength);
091                                ViewTransform view = new ViewTransform(w);
092                                for (int j = 0; j < N; j++) {
093                                        double[] uv = cam.project(view, modelPts[j]);
094                                        Y[l * 2 + 0] = uv[0];
095                                        Y[l * 2 + 1] = uv[1];
096                                        l = l + 1;
097                                }
098                        }
099                        return Y;
100                }
101        }       // end of inner class 'ValueFun'
102
103        // ---------------------------------------------------------------------
104
105        private RealVector makeInitialParameters(Camera camera, ViewTransform[] views) {
106                double[] s = camera.getParameterVector();
107                double[] c = new double[s.length + M * viewParLength];
108
109                // insert camera parameters at beginning of c
110                System.arraycopy(s, 0, c, 0, s.length);
111
112                // insert M view parameters
113                int start = s.length;
114                for (int i = 0; i < M; i++) {
115                        double[] w = views[i].getParameters();
116                        System.arraycopy(w, 0, c, start, w.length);
117                        start = start + w.length;
118                }
119                return new ArrayRealVector(c);
120        }
121
122
123        /**
124         * Stack the observed image coordinates of the calibration pattern points into
125         * a vector.
126         * @return the observed vector
127         */
128        protected RealVector makeObservedVector()       {
129                double[] obs = new double[M * N * 2];
130                for (int i = 0, k = 0; i < M; i++) {
131                        for (int j = 0; j < N; j++, k++) {
132                                obs[k * 2 + 0] = obsPts[i][j].getX();
133                                obs[k * 2 + 1] = obsPts[i][j].getY();
134                        }
135                }
136                // obs = [u_{0,0}, v_{0,0}, u_{0,1}, v_{0,1}, ..., u_{M-1,N-1}, v_{M-1,N-1}]
137                return new ArrayRealVector(obs); 
138        }
139
140        private void updateEstimates(RealVector parameters) {
141                double[] c = parameters.toArray();
142                double[] s = Arrays.copyOfRange(c, 0, camParLength);
143                finalCamera = new Camera(s);
144
145                finalViews = new ViewTransform[M];
146                int start = s.length;
147                for (int i = 0; i < M; i++) {
148                        double[] w = Arrays.copyOfRange(c, start, start + viewParLength);
149                        finalViews[i] = new ViewTransform(w);
150                        start = start + w.length;
151                }
152        }
153        
154        /**
155         * Returns the optimized camera parameters.
156         * @return the optimized camera parameters
157         */
158        protected Camera getFinalCamera() {
159                return finalCamera;
160        }
161
162        /**
163         * Returns the optimized view parameters.
164         * @return the optimized view parameters
165         */
166        protected ViewTransform[] getFinalViews() {
167                return finalViews;
168        }
169        
170}
