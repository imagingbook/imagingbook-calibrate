039                @Override
040                public double[][] value(double[] params) {
041                        final double[][] J = new double[2 * M * N][];
042                        for (int i = 0, k = 0; i < M; i++) {
043                                for (int j = 0; j < N; j++, k++) {
044                                        final double[][] tmp = computeJacobian(i, j, params);
045                                        J[k * 2 + 0] = tmp[0];
046                                        J[k * 2 + 1] = tmp[1];
047                                }
048                        }
049                        return J;
050                }
