035                public double[][] value(double[] params) {
036                        final double[][] J = new double[2 * M * N][];
037                        for (int i = 0, k = 0; i < M; i++) {
038                                for (int j = 0; j < N; j++, k++) {
039                                        final double[][] tmp = computeJacobian(i, j, params);
040                                        J[k * 2 + 0] = tmp[0];
041                                        J[k * 2 + 1] = tmp[1];
042                                }
043                        }
044                        return J;
045                }
