040        /**
041         * Estimates the homographies between a fixed set of 2D model points and
042         * multiple observations (image point sets).
043         * The correspondence between the points is assumed to be known.
044         * @param modelPts a sequence of 2D points on the model (calibration target)
045         * @param obsPoints a sequence 2D image point sets (one set per view).
046         * @return the sequence of estimated homographies (3 x 3 matrices), one for each view
047         */
048        public RealMatrix[] estimateHomographies(Point2D[] modelPts, Point2D[][] obsPoints) {
049                final int M = obsPoints.length;
050                RealMatrix[] homographies = new RealMatrix[M];
051                for (int i = 0; i < M; i++) {
052                        RealMatrix Hinit = estimateHomography(modelPts, obsPoints[i]);
053                        RealMatrix H = refineHomography(Hinit, modelPts, obsPoints[i]);
054                        homographies[i] = H;
055                }
056                return homographies;
057        }
058        
059        /**
060         * Estimates the homography (projective) transformation from two given 2D point sets.
061         * The correspondence between the points is assumed to be known.
062         * @param ptsA the 1st sequence of 2D points 
063         * @param ptsB the 2nd sequence of 2D points
064         * @return the estimated homography (3 x 3 matrix)
065         */
066        public RealMatrix estimateHomography(Point2D[] ptsA, Point2D[] ptsB) {
067                int n = ptsA.length;
068                
069                RealMatrix Na = (normalizePointCoordinates) ? getNormalisationMatrix(ptsA) : MatrixUtils.createRealIdentityMatrix(3);
070                RealMatrix Nb = (normalizePointCoordinates) ? getNormalisationMatrix(ptsB) : MatrixUtils.createRealIdentityMatrix(3);   
071                RealMatrix M = MatrixUtils.createRealMatrix(n * 2, 9);
072
073                for (int j = 0, r = 0; j < ptsA.length; j++) {
074                        final double[] pA = transform(MathUtil.toArray(ptsA[j]), Na);
075                        final double[] pB = transform(MathUtil.toArray(ptsB[j]), Nb);
076                        final double xA = pA[0];
077                        final double yA = pA[1];
078                        final double xB = pB[0];
079                        final double yB = pB[1];                        
080                        M.setRow(r + 0, new double[] {xA, yA, 1, 0, 0, 0, -(xA * xB), -(yA * xB), -(xB)});
081                        M.setRow(r + 1, new double[] {0, 0, 0, xA, yA, 1, -(xA * yB), -(yA * yB), -(yB)});
082                        r = r + 2;
083                }
084
085                // find h, such that M . h = 0:
086                double[] h = MathUtil.solveHomogeneousSystem(M).toArray();
087                
088                // assemble homography matrix H from h:
089                RealMatrix H = MatrixUtils.createRealMatrix(new double[][] 
090                                {{h[0], h[1], h[2]},
091                                 {h[3], h[4], h[5]},
092                                 {h[6], h[7], h[8]}} );
093
094                // de-normalize the homography
095                H = MatrixUtils.inverse(Nb).multiply(H).multiply(Na);
096                
097                // rescale M such that H[2][2] = 1 (unless H[2][2] close to 0)
098                if (Math.abs(H.getEntry(2, 2)) > 10e-8) {
099                        H = H.scalarMultiply(1.0 / H.getEntry(2, 2));
100                }
101                return H;
102        }

105        /**
106         * Refines the initial homography by non-linear (Levenberg-Marquart) optimization.
107         * @param Hinit the initial (estimated) homography matrix
108         * @param pntsA the 1st sequence of 2D points 
109         * @param pntsB the 2nd sequence of 2D points
110         * @return the refined homography matrix
111         */
112        public RealMatrix refineHomography(RealMatrix Hinit, Point2D[] pntsA, Point2D[] pntsB) {
113                final int M = pntsA.length;             
114                double[] observed = new double[2 * M];
115                for (int i = 0; i < M; i++) {
116                        observed[i * 2 + 0] = pntsB[i].getX();
117                        observed[i * 2 + 1] = pntsB[i].getY();
118                }                       
119                MultivariateVectorFunction value = getValueFunction(pntsA);
120                MultivariateMatrixFunction jacobian = getJacobianFunction(pntsA);
121                
122                LeastSquaresProblem problem = LeastSquaresFactory.create(
123                                LeastSquaresFactory.model(value, jacobian),
124                                MatrixUtils.createRealVector(observed), 
125                                MathUtil.getRowPackedVector(Hinit), 
126                                null,  // ConvergenceChecker
127                                maxLmEvaluations, 
128                                maxLmIterations);
129                
130                LevenbergMarquardtOptimizer lm = new LevenbergMarquardtOptimizer();
131                Optimum result = lm.optimize(problem);
132                
133                RealVector optimum = result.getPoint();
134                RealMatrix Hopt = MathUtil.fromRowPackedVector(optimum, 3, 3);
135//              System.out.println("LM optimizer iterations " + result.getIterations());
136                return Hopt.scalarMultiply(1.0 / Hopt.getEntry(2, 2));
137        }

140        private MultivariateVectorFunction getValueFunction(Point2D[] X) {
141                //System.out.println("MultivariateVectorFunction getValueFunction");
142                return new MultivariateVectorFunction() {
143                        public double[] value(double[] h) {
144                                final double[] Y = new double[X.length * 2];
145                                for (int j = 0; j < X.length; j++) {
146                                        final double x = X[j].getX();
147                                        final double y = X[j].getY();
148                                        final double w = h[6] * x + h[7] * y + h[8];
149                                        Y[j * 2 + 0] = (h[0] * x + h[1] * y + h[2]) / w;
150                                        Y[j * 2 + 1] = (h[3] * x + h[4] * y + h[5]) / w;
151                                }
152                                return Y;
153                        }
154                };
155        }
156        
157        protected MultivariateMatrixFunction getJacobianFunction(Point2D[] X) {
158                return new MultivariateMatrixFunction() {
159                        public double[][] value(double[] h) {
160                                final double[][] J = new double[2 * X.length][];
161                                for (int i = 0; i < X.length; i++) {
162                                        final double x = X[i].getX();
163                                        final double y = X[i].getY();
164                                        
165                                        final double w  = h[6] * x + h[7] * y + h[8];
166                                        final double w2 = w * w;
167                                        
168                                        final double sx = h[0] * x + h[1] * y + h[2];           
169                                        J[2 * i + 0] = new double[] {x/w, y/w, 1/w, 0, 0, 0, -sx*x/w2, -sx*y/w2, -sx/w2};
170                                        
171                                        final double sy = h[3] * x + h[4] * y + h[5];
172                                        J[2 * i + 1] = new double[] {0, 0, 0, x/w, y/w, 1/w, -sy*x/w2, -sy*y/w2, -sy/w2};
173                                }
174                                return J;
175                        }
176                };
177        }

272        private double[] transform(double[] p, RealMatrix M3x3) {
273                double[] pA = MathUtil.toHomogeneous(p);
274                double[] pAt = M3x3.operate(pA);
275                return MathUtil.toCartesian(pAt); // need to de-homogenize, since pAt[2] == 1?
276        }
277        
278        private RealMatrix getNormalisationMatrix(Point2D[] pnts) {
279                final int N = pnts.length;
280                double[] x = new double[N];
281                double[] y = new double[N];
282                
283                for (int i = 0; i < N; i++) {
284                        x[i] = pnts[i].getX();
285                        y[i] = pnts[i].getY();
286                }
287                
288                // calculate the means in x/y
289                double meanx = MathUtil.mean(x);
290                double meany = MathUtil.mean(y);
291
292                // calculate the variances in x/y
293                double varx = MathUtil.variance(x);
294                double vary = MathUtil.variance(y);
295                
296                double sx = Math.sqrt(2 / varx);
297                double sy = Math.sqrt(2 / vary);
298
299                RealMatrix matrixA = MatrixUtils.createRealMatrix(new double[][] {
300                                { sx,  0, -sx * meanx},
301                                {  0, sy, -sy * meany},
302                                {  0,  0,           1 }});
303                
304                return matrixA;
305        }

310        static Random rand = new Random();
311        private static Point2D apply(RealMatrix H, Point2D X, double noise) {
312                
313                double[] Xa = {X.getX(), X.getY(), 1};
314                double[] Xb = H.operate(Xa);
315                double xn = noise * rand.nextGaussian();
316                double yn = noise * rand.nextGaussian();
317                return new Point2D.Double(xn + Xb[0]/Xb[2], yn + Xb[1]/Xb[2]);
318        }
