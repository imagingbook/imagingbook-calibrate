003import java.io.StringWriter;
004import java.util.Arrays;
005
006import org.apache.commons.math3.geometry.euclidean.threed.Rotation;
007import org.apache.commons.math3.geometry.euclidean.threed.RotationConvention;
008import org.apache.commons.math3.geometry.euclidean.threed.Vector3D;
009import org.apache.commons.math3.linear.MatrixUtils;
010import org.apache.commons.math3.linear.RealMatrix;
011import org.apache.commons.math3.linear.RealVector;
012
013import imagingbook.lib.math.Matrix;
014
015/**
016 * Instances of this class represent extrinsic camera (view) parameters.
017 * @author WB
018 *
019 */
020public class ViewTransform {
021        
022        private static final double OrthogonalityThreshold = 0.01;
023        
024        private final Rotation rotation;
025        private final double[] translation;
026        
027        // ----------------------------------------------------------------------------------
028        
029        public ViewTransform() {
030                this.rotation = Rotation.IDENTITY;
031                this.translation = new double[3];
032        }
033        
034        public ViewTransform(double rX, double rY, double rZ, double tX, double tY, double tZ) {
035                this.rotation = makeRotation(new double[] {rX, rY, rZ});
036                this.translation = new double[] {tX, tY, tZ};
037        }
038        
039        public ViewTransform(Rotation rot, double[] t) {
040                this.rotation = rot;
041                translation = t;
042        }
043        
044        public ViewTransform(RealMatrix RT) {   // RT is of size 3 x 4 (a homography)
045                if (RT.getRowDimension() != 3 || RT.getColumnDimension() != 4) {
046                        throw new IllegalArgumentException("View transform matrix must be 3 x 4");
047                }
048                RealMatrix R = RT.getSubMatrix(0, 2, 0, 2);
049                rotation = new Rotation(R.getData(), OrthogonalityThreshold);
050                translation = RT.getColumnVector(3).toArray();
051        }
052        
053        public ViewTransform(RealMatrix R, RealVector t) {      // R is of size 3 x 3 , t of size 3 x 1
054                this(new Rotation(R.getData(), OrthogonalityThreshold), t.toArray());
055//              if (R.getRowDimension() != 3 || R.getColumnDimension() != 3) {
056//                      throw new IllegalArgumentException("Rotation matrix must be 3 x 3");
057//              }
058//              if (t.getDimension() != 3) {
059//                      throw new IllegalArgumentException("Translation vector must be of dimension 3");
060//              }
061//              rotation = new Rotation(R.getData(), OrthogonalityThreshold);
062//              translation = t.toArray();
063        }
064        
065        public ViewTransform(double[] w) {
066                this.rotation = makeRotation(w);
067                this.translation = Arrays.copyOfRange(w, 3, 6);
068        }
069        
070        // ----------------------------------------------------------------------------------
071        
072        private Rotation makeRotation(double[] w) {
073                Vector3D axis = new Vector3D(w[0], w[1], w[2]);
074                double angle = axis.getNorm();
075                //return new Rotation(axis, angle);
076                return new Rotation(axis, angle, RotationConvention.VECTOR_OPERATOR);
077                 
078        }
079                
080        protected int getParameterLength() {
081                return 6;
082        }
083        
084        protected double[] getParameters() {
085                //double[] rotAxis = rotation.getAxis().toArray();
086                double[] rotAxis = rotation.getAxis(RotationConvention.VECTOR_OPERATOR).toArray();
087                double rotAngle = rotation.getAngle();
088                return new double[] {
089                        rotAxis[0] * rotAngle,
090                        rotAxis[1] * rotAngle,
091                        rotAxis[2] * rotAngle,
092                        translation[0], translation[1], translation[2]};
093        }
094        
095        public Rotation getRotation() {
096                return rotation;
097        }
098        
099        public double[] getRotationAxis() {
100                //double[] rotAxis = rotation.getAxis().toArray();
101                double[] rotAxis = rotation.getAxis(RotationConvention.VECTOR_OPERATOR).toArray();
102                double rotAngle = rotation.getAngle();
103                rotAxis[0] *= rotAngle;
104                rotAxis[1] *= rotAngle;
105                rotAxis[2] *= rotAngle;
106                return rotAxis;
107        }
108        
109        public RealMatrix getRotationMatrix() {
110                double[][] R = rotation.getMatrix();
111                return MatrixUtils.createRealMatrix(R);
112        }
113        
114        public double[] getTranslation() {
115                return translation;
116        }
117        
118        public RealVector getTranslationVector() {
119                return MatrixUtils.createRealVector(translation);
120        }
121        
122        // ----------------------------------------------------------------------------------
123        
124        /**
125         * Moves point XYZ from 3D world coordinates to 3D camera coordinates,
126         * as specified by the transformations of this view.
127         * No projection is involved.
128         * @param XYZ a 3D (world) point
129         * @return the given world point mapped to 3D camera coordinates
130         */
131        protected double[] applyTo(double[] XYZ) {      // 3D vector XYZ assumed
132                double[] XYZc = new double[3];
133                rotation.applyTo(XYZ, XYZc);
134                for (int i = 0; i < 3; i++) {
135                        XYZc[i] = XYZc[i] + translation[i];
136                }
137                return XYZc;
138        }
139        
140        public String toString() {
141                RealMatrix R = this.getRotationMatrix();
142                RealVector T = this.getTranslationVector();
143                StringWriter writer = new StringWriter();
144                writer.append("R = \n");
145                writer.append(Matrix.toString(R.getData()));
146                writer.append("\n");
147                writer.append("T = ");
148                writer.append(Matrix.toString(T.toArray()));
149                return writer.toString();
150        }
151        
152        // ------------------------------------------------------------------
153        
154        /**
155         * For testing only.
156         * @param args ignored
157         */
158        public static void main(String[] args) {
159                double[] w1 = {1.2, -0.5, 0.9, 10, 20, 30};
160                System.out.println("w1 = " + Matrix.toString(w1));
161                ViewTransform view = new ViewTransform(w1);
162                double[] w2 = view.getParameters();
163                System.out.println("w2 = " + Matrix.toString(w2));
164        }
