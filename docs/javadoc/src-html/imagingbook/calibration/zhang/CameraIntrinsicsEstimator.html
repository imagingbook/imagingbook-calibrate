020        /**
021         * Version 1 (Zhang's original closed form solution).
022         * Estimates the the intrinsic camera parameters from multiple homographies.
023         * @param homographies a set of homography matrices
024         * @return the estimated 3 x 3 intrinsic transformation matrix
025         */
026        protected RealMatrix getCameraIntrinsicsZhang1(RealMatrix[] homographies) {
027                final int M = homographies.length;
028                int rows = 2 * M;
029                double[][] V = new double[rows][];
030
031                for (int i = 0; i < M; i++) {
032                        RealMatrix H = homographies[i];
033                        V[2*i] = getVpq(H, 0, 1); // v01
034                        V[2*i + 1] = Matrix.subtract(getVpq(H, 0, 0), getVpq(H, 1, 1)); // v00-v11
035                }
036
037                if (M == 2) {
038                        V[V.length - 1] = new double[] { 0, 1, 0, 0, 0, 0 };
039                }
040                
041                RealMatrix VM = MatrixUtils.createRealMatrix(V);
042//              MathUtil.print("estimateIntrinsics: V = ", VM);//WB
043                
044                double[] b = MathUtil.solveHomogeneousSystem(VM).toArray();     // solve VM.b=0
045                
046                final double vc         = (b[1] * b[3] - b[0] * b[4]) / (b[0] * b[2] - b[1] * b[1]);
047                final double lambda = b[5] - (b[3] * b[3] + vc * (b[1] * b[3] - b[0] * b[4])) / b[0];
048                final double alpha      = Math.sqrt(lambda / b[0]);
049                final double beta       = Math.sqrt(lambda * b[0] / (b[0] * b[2] - b[1] * b[1]));
050                final double gamma      = -b[1] * alpha * alpha * beta / lambda;
051                final double uc         = gamma * vc / alpha - b[3] * alpha * alpha / lambda;   // PAMI paper = WRONG!
052
053                RealMatrix A = MatrixUtils.createRealMatrix(new double[][] {
054                                { alpha, gamma, uc },
055                                { 0, beta, vc },
056                                { 0, 0, 1 }
057                });
058
059                return A;
060        }

066         * @return the estimated 3 x 3 intrinsic transformation matrix
067         */

069                final int M = homographies.length;

074                        RealMatrix H = homographies[i];
075                        V[2*i] = getVpq(H, 0, 1); // v01
076                        V[2*i + 1] = Matrix.subtract(getVpq(H, 0, 0), getVpq(H, 1, 1)); // v00-v11
077                }
078
079                if (M == 2) {
080                        V[V.length - 1] = new double[] { 0, 1, 0, 0, 0, 0 };
081                }
082                
083                RealMatrix VM = MatrixUtils.createRealMatrix(V);
084                double[] b = MathUtil.solveHomogeneousSystem(VM).toArray();     // solve VM.b=0
085                
086                final double vc         = (b[1] * b[3] - b[0] * b[4]) / (b[0] * b[2] - b[1] * b[1]);
087                final double lambda = b[5] - (b[3] * b[3] + vc * (b[1] * b[3] - b[0] * b[4])) / b[0];
088                final double alpha      = Math.sqrt(lambda / b[0]);

093                RealMatrix A = MatrixUtils.createRealMatrix(new double[][] {
094                                { alpha, gamma, uc },
095                                { 0, beta, vc },

102        /**
103         * Version 3 (WB's closed form solution).
104         * Estimates the the intrinsic camera parameters from multiple homographies.
105         * @param homographies a set of homography matrices
106         * @return the estimated 3 x 3 intrinsic transformation matrix
107         */
108        protected RealMatrix getCameraIntrinsicsZhang3(RealMatrix[] homographies) {
109                final int M = homographies.length;
110                int rows = 2 * M;
111                double[][] V = new double[rows][];
112
113                for (int i = 0; i < M; i++) {
114                        RealMatrix H = homographies[i];
115                        V[2*i] = getVpq(H, 0, 1); // v01
116                        V[2*i + 1] = Matrix.subtract(getVpq(H, 0, 0), getVpq(H, 1, 1)); // v00-v11
117                }
118
119                if (M == 2) {
120                        V[V.length - 1] = new double[] { 0, 1, 0, 0, 0, 0 };
121                }
122                
123                RealMatrix VM = MatrixUtils.createRealMatrix(V);
124                double[] b = MathUtil.solveHomogeneousSystem(VM).toArray();     // solve VM.b=0
125                
126                final double w = b[0]*b[2]*b[5] - b[1]*b[1]*b[5] - b[0]*b[4]*b[4] + 2*b[1]*b[3]*b[4] - b[2]*b[3]*b[3];
127                final double d = 
128                                b[0] * b[2] - b[1] * b[1];
129                final double uc = 
130                                (b[1] * b[4] - b[2] * b[3]) / d;
131                final double vc = 
132                                (b[1] * b[3] - b[0] * b[4]) / d;
133                final double alpha      = 
134                                Math.sqrt(w / (d * b[0]));
135                final double beta       = 
136                                Math.sqrt(w / (d * d) * b[0]);
137                final double gamma      = 
138                                  Math.sqrt(w / (d * d * b[0])) * b[1];
139                
140                RealMatrix A = MatrixUtils.createRealMatrix(new double[][] {
141                                { alpha, gamma, uc },
142                                { 0, beta, vc },
143                                { 0, 0, 1 }
144                });
145
146                return A;
147        }

150        /**
151         * Final version by WB (this version is used by default).
152         * Estimates the the intrinsic camera parameters from multiple homographies
153         * using a Cholesky decomposition.
154         * @param homographies a set of homography matrices
155         * @return the estimated 3 x 3 intrinsic transformation matrix
156         */
157        protected RealMatrix getCameraIntrinsics(RealMatrix[] homographies) {
158                final int M = homographies.length;
159                int rows = 2 * M;
160                double[][] V = new double[rows][];
161
162                for (int i = 0; i < M; i++) {
163                        RealMatrix H = homographies[i];
164                        V[2*i] = getVpq(H, 0, 1); // v01
165                        V[2*i + 1] = Matrix.subtract(getVpq(H, 0, 0), getVpq(H, 1, 1)); // v00-v11
166                }
167
168                if (M == 2) {
169                        V[V.length - 1] = new double[] { 0, 1, 0, 0, 0, 0 };
170                }
171                
172                RealMatrix VM = MatrixUtils.createRealMatrix(V);
173//              MathUtil.print("estimateIntrinsics: V = ", VM);//WB
174                
175                double[] b = MathUtil.solveHomogeneousSystem(VM).toArray();     // solve VM.b=0
176                
177                RealMatrix B = MatrixUtils.createRealMatrix(new double[][]
178                                {{b[0], b[1], b[3]},
179                                 {b[1], b[2], b[4]},
180                                 {b[3], b[4], b[5]}});
181                
182                if (B.getEntry(0, 0) < 0 || B.getEntry(1, 1) < 0 || B.getEntry(2, 2) < 0) {     
183                        B = B.scalarMultiply(-1);       // make sure B is positive definite 
184                }
185                
186                CholeskyDecomposition cd = new CholeskyDecomposition(B);
187                RealMatrix L = cd.getL();
188                RealMatrix A = MatrixUtils.inverse(L).transpose().scalarMultiply(L.getEntry(2, 2));
189                return A;
190        }

205        // version without transpose
206        private double[] getVpq(RealMatrix H, int p, int q) {
207                final double[] vpq = new double[] {
208                                H.getEntry(0, p) * H.getEntry(0, q),
209                                H.getEntry(0, p) * H.getEntry(1, q) + H.getEntry(1, p) * H.getEntry(0, q),
210                                H.getEntry(1, p) * H.getEntry(1, q),
211                                H.getEntry(2, p) * H.getEntry(0, q) + H.getEntry(0, p) * H.getEntry(2, q),
212                                H.getEntry(2, p) * H.getEntry(1, q) + H.getEntry(1, p) * H.getEntry(2, q),
213                                H.getEntry(2, p) * H.getEntry(2, q)
214                };
215                return vpq;
216        }
