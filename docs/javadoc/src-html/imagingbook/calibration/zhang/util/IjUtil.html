015/**
016 * This class defines ImageJ-related utility methods used for camera
017 * calibration.
018 * @author WB
019 *
020 */
021public class IjUtil {
022        
023        /**
024         * Creates a float-type image of the supplied matrix.
025         * Matrix rows/columns correspond to image rows/columns.
026         * Typical usage: makeImage("my matrix", M).show()
027         * @param title the image title
028         * @param M the matrix to be displayed
029         * @return the resulting image
030         */
031        public static ImagePlus makeImage(String title, RealMatrix M) {
032                float[][] fdata = Matrix.duplicateToFloat(M.transpose().getData());
033                ImagePlus im = new ImagePlus(title, new FloatProcessor(fdata));
034                im.show();
035                return im;      
036        }
037        
038        // -------------------------------------------------------------------
039        
040        
041        public static PlotWindow plotLensDistortionFunction(Camera cam, double rmax) {
042                int n = 100;
043//              double[] K = cam.getK();
044                double[] xVals = new double[n];
045                double[] yVals = new double[n];
046                for (int i = 0; i < n; i++) {
047                        double r = i * (rmax / n);
048                        //double d = cam.distFun2(r * r); //getRadialDistortion(r, K);
049                        double d = cam.D(r);
050                        xVals[i] = r;
051                        yVals[i] = d;
052                }
053                Plot plot = new Plot("d(r)", "x", "f(x)");
054                plot.setColor(Color.BLUE);
055                plot.setLimits(0, 1.25, -0.1, 0.1);
056                plot.addPoints(xVals, yVals, Plot.LINE);
057                plot.draw();
058                return plot.show();
059        }
