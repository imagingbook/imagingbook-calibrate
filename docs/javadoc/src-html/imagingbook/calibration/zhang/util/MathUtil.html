072        public static RealVector crossProduct3x3(RealVector A, RealVector B) {
073                final double[] a = A.toArray();
074                final double[] b = B.toArray();
075                final double[] c = {
076                                a[1] * b[2] - b[1] * a[2],
077                                a[2] * b[0] - b[2] * a[0],
078                                a[0] * b[1] - b[0] * a[1]
079                };
080                return MatrixUtils.createRealVector(c);
081        }

087        public static RealVector getRowPackedVector(RealMatrix A) {
088                double[][] AA = A.getData();
089                double[] V = new double[AA.length * AA[0].length];
090                int k = 0;
091                for (int i = 0; i < AA.length; i++) {
092                        for (int j = 0; j < AA[0].length; j++) {
093                                V[k++] = AA[i][j];
094                        }
095                }
096                return MatrixUtils.createRealVector(V);
097        }
098        
099        public static RealMatrix fromRowPackedVector(RealVector V, int rows, int columns) {
100                double[][] AA = new double[rows][columns];
101                double[] data = V.toArray();
102                int k = 0;
103                for (int i = 0; i < rows; i++) {
104                        for (int j = 0; j < columns; j++) {
105                                AA[i][j] = data[k++];
106                        }
107                }
108                return MatrixUtils.createRealMatrix(AA);
109        }

154        /**               
155         * Finds a nontrivial solution (x) to the homogeneous linear system M . x = 0
156         * by singular-value decomposition. If M has more rows than columns, the
157         * system of equations is overdetermined. In this case the returned solution
158         * minimizes the residual ||M . x|| in the least-squares sense.
159         * 
160         * @param A     the original matrix.
161         * @return the solution vector x.
162         */
163        public static RealVector solveHomogeneousSystem(RealMatrix A) {
164                SingularValueDecomposition svd = new SingularValueDecomposition(A);
165                RealMatrix V = svd.getV();
166                RealVector x = V.getColumnVector(V.getColumnDimension() - 1);
167                return x;
168        }

170        public static double[] toHomogeneous(double[] cvec) {
171                double[] hvec = new double[cvec.length + 1];
172                for (int i = 0; i < cvec.length; i++) {
173                        hvec[i] = cvec[i];
174                        hvec[hvec.length - 1] = 1;
175                }
176                return hvec;
177        }

179        public static double[] toCartesian(double[] hvec) {
180                double[] cvec = new double[hvec.length - 1];
181                final double s = 1 / hvec[hvec.length - 1];     // TODO: check for zero factor
182                for (int i = 0; i < hvec.length - 1; i++) {
183                        cvec[i] = s * hvec[i];
184                }
185                return cvec;
186        }

223        public static double mean(double[] x) {
224                final int n = x.length;
225                if (n == 0) 
226                        return 0;
227                double sum = 0;
228                for (int i = 0; i < x.length; i++) {
229                        sum = sum + x[i];
230                }
231                return sum / n;
232        }
233        
234        /**
235         * 
236         * @param x a sequence of real values 
237         * @return the variance of the values in x (sigma^2)
238         */
239        public static double variance(double[] x) {
240                final int n = x.length;
241                if (n == 0) 
242                        return 0;
243                double sum = 0;
244                double sum2 = 0;
245                for (int i = 0; i < x.length; i++) {
246                        sum = sum + x[i];
247                        sum2 = sum2 + x[i] * x[i];
248                }
249                return (sum2 - (sum * sum) / n) / n;
250        }

289        /**
290         * Calculates the 'inverse condition number' (RCOND() in Matlab)
291         * of the given matrix (0,...,1, ideally close to 1).
292         * @param Ma the matrix
293         * @return the inverse condition number
294         */
295        public static double inverseConditionNumber(double[][] Ma) {
296                RealMatrix M = new Array2DRowRealMatrix(Ma);
297                SingularValueDecomposition svd = new SingularValueDecomposition(M);
298                return svd.getInverseConditionNumber();
299        }
