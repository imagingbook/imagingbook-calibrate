001package calibration_demos;
002
003import java.awt.Color;
004import java.awt.geom.Point2D;
005
006import ij.IJ;
007import ij.ImagePlus;
008import ij.ImageStack;
009import ij.plugin.PlugIn;
010import ij.process.ImageProcessor;
011import imagingbook.calibration.zhang.Camera;
012import imagingbook.calibration.zhang.ViewTransform;
013import imagingbook.calibration.zhang.testdata.ZhangData;
014import imagingbook.calibration.zhang.util.GridPainter;
015import imagingbook.lib.ij.IjLogStream;
016import imagingbook.lib.settings.PrintPrecision;
017import imagingbook.lib.util.ResourceUtils;
018
019
020/**
021 * This plugin projects opens an image stack containing the 5 Zhang
022 * test images and projects the model points into each view,
023 * using the (known) camera and view parameters.
024 * All data are part of Zhang's demo data set that comes with the
025 * EasyCalib program. No calibration is performed.
026 * 
027 * @author W. Burger
028 * @version 2017-05-30
029 */
030public class Demo_Zhang_Projection implements PlugIn {
031        
032        static Class<?> resourceRootClass = ZhangData.class;
033        static String resourceDir = "resources/";
034        static String resourceName = "CalibImageStack.tif";
035        
036        static Color BackGroundColor = Color.white;
037        static Color LineColor = Color.magenta;
038        static boolean BeVerbose = false;
039        
040        static {
041                IjLogStream.redirectSystem();
042                PrintPrecision.set(6);
043        }
044        
045
046        public void run(String arg0) {
047                // open the test image (stack):
048                ImagePlus testIm = ResourceUtils.openImageFromResource(resourceRootClass, resourceDir, resourceName);
049                if (testIm == null) {
050                        IJ.error("Could not open calibration images!");
051                        return;
052                }
053                testIm.show();
054                
055                // get the 3D model points:
056                Point2D[] modelPoints = ZhangData.getModelPoints();
057                
058                // get the camera intrinsics (typically by calibration):
059                Camera camera = ZhangData.getCameraIntrinsics();
060                
061                // get all view parameters (extrinsics):
062                ViewTransform[] views = ZhangData.getAllViewTransforms();
063                int M = views.length;
064
065                // project and draw the model into the views:
066                ImageStack stack = testIm.getStack();
067                for (int i = 0; i < M; i++) {
068                        // project the model points:
069                        Point2D[] projectedPoints = camera.project(views[i], modelPoints);
070                        ImageProcessor ip = stack.getProcessor(i + 1);
071                        // draw the squares:
072                        GridPainter painter = new GridPainter(ip);
073                        painter.lineCol = LineColor;
074                        painter.drawSquares(projectedPoints);
075                }
076                testIm.updateAndDraw();
077        }
078
079}
