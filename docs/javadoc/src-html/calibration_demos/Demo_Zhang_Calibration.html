001package calibration_demos;
002
003import java.awt.Color;
004import java.awt.geom.Point2D;
005
006import ij.IJ;
007import ij.ImagePlus;
008import ij.ImageStack;
009import ij.plugin.PlugIn;
010import ij.process.ColorProcessor;
011import ij.process.ImageProcessor;
012import imagingbook.calibration.zhang.Calibrator;
013import imagingbook.calibration.zhang.Calibrator.Parameters;
014import imagingbook.calibration.zhang.Camera;
015import imagingbook.calibration.zhang.ViewTransform;
016import imagingbook.calibration.zhang.testdata.ZhangData;
017import imagingbook.calibration.zhang.util.GridPainter;
018import imagingbook.lib.ij.IjLogStream;
019import imagingbook.lib.settings.PrintPrecision;
020import imagingbook.lib.util.ResourceUtils;
021
022/**
023 * This plugin performs Zhang's camera calibration on the
024 * pre-calculated point data for the N given target views.
025 * Based on the estimated intrinsic and extrinsic (view)
026 * parameters, the corner points of the 3D target model are
027 * then projected onto the corresponding calibration images
028 * (a stack).
029 * All rendering is done by pixel drawing (no graphic overlays).
030 * 
031 * @author W. Burger
032 * @version 2018/12/29
033 */
034public class Demo_Zhang_Calibration implements PlugIn {
035        
036        private static boolean DEBUG = true;
037        
038        private static Class<?> resourceRootClass = ZhangData.class;
039        private static String resourceDir = "resources/";
040        private static String resourceName = "CalibImageStack.tif";
041        
042        private static boolean ShowObservedModelPoints = true;          // draw observed image points into a new stack
043        private static boolean ShowProjectedImagePoints = true;         // draw projected image points into the test image stack
044        private static boolean ListCameraViews = true;
045        
046        private static Color BackGroundColor = Color.white;
047        
048        static {
049                IjLogStream.redirectSystem();
050                PrintPrecision.set(6);
051        }
052        
053        public void run(String arg0) {
054                ImagePlus testIm = ResourceUtils.openImageFromResource(resourceRootClass, resourceDir, resourceName);
055                if (testIm == null) {
056                        IJ.error("Could not open calibration images!"); 
057                        return;
058                }
059                
060                int M = testIm.getNSlices();    // number of views
061                if (M < 2) {
062                        IJ.error("Image must be a stack with 2+ images!"); 
063                        return;
064                }
065                
066                testIm.show();
067                int width = testIm.getWidth();
068                int height = testIm.getHeight();
069                Point2D[] modelPoints = ZhangData.getModelPoints();
070                Camera camReal = ZhangData.getCameraIntrinsics();
071                if (DEBUG) {
072                        System.out.println("Camera intrinsics (reference from EasyCalib):");
073                        System.out.println(camReal.toString());
074                }
075                
076//              ViewTransform[] viewsReal = ZhangData.getAllViewTransforms();   
077                Point2D[][] obsPoints     = ZhangData.getAllObservedPoints();
078                        
079                if (ShowObservedModelPoints){
080                        ImageStack stack = new ImageStack(width, height);
081                        drawSquares(stack, obsPoints);
082                        new ImagePlus("Observed points", stack).show();
083                }
084                
085                // Set up calibrator ------------------------------------------
086                
087                Parameters params = new Calibrator.Parameters();
088                params.normalizePointCoordinates = true;
089                params.lensDistortionKoeffients = 2;
090                params.useNumericJacobian = true;
091                params.debug = DEBUG;
092                
093                Calibrator zcalib = new Calibrator(params, modelPoints);
094                for (int i = 0; i < M; i++) {
095                        zcalib.addView(obsPoints[i]);
096                }
097                
098                // Perform calibration ------------------------------------------
099                
100                Camera camFinal = zcalib.calibrate();
101                if (camFinal == null) {
102                        System.out.println("calibration failed");
103                        return;
104                }
105                
106                // Show results ------------------------------------------
107                
108                System.out.println("Camera intrinsics (final estimate):");
109                System.out.println(camFinal.toString());
110                
111                ViewTransform[] finalViews = zcalib.getFinalViews();
112                System.out.format("Final (squared) projection error: %.3f\n", zcalib.getProjectionError(camFinal, finalViews, obsPoints));
113                                
114                if (ListCameraViews) {
115                        for (int i = 0; i < M; i++) {
116                                System.out.println("**** View " + i + ": ****");
117                                System.out.println(finalViews[i].toString());
118                        }
119                }
120                
121                if (ShowProjectedImagePoints) {
122                        Point2D[][] projPoints = new Point2D[M][];
123                        for (int i = 0; i < M; i++) {
124                                projPoints[i] = camFinal.project(finalViews[i], modelPoints);
125                        }
126                        drawSquares(testIm.getStack(), projPoints);
127                        testIm.updateAndDraw();
128                }
129        }
130        
131        // --------------------------------------------------------------------
132        
133        /**
134         * Draws the array of image points to a given (possibly empty) stack image.
135         * The image points are assumed to be the corners of the standard
136         * calibration model, i.e., 4 consecutive points form a projected square.
137         * @param stack a stack with M images (views)
138         * @param imagePoints a sequence of 2D point sets, one for each view
139         */
140        private void drawSquares(ImageStack stack, Point2D[][] imagePoints) {
141                final int width = stack.getWidth();
142                final int height = stack.getHeight();
143                int M = imagePoints.length;
144                for (int i = 0; i < M; i++) {
145                        ImageProcessor ip;
146                        if (stack.getSize() > i) {      // use existing stack slice
147                                ip = stack.getProcessor(i + 1);
148                        }
149                        else {                                          // create and fill new slice
150                                ip = new ColorProcessor(width, height);
151                                if (BackGroundColor != null) {
152                                        ip.setColor(BackGroundColor);
153                                        ip.fill();
154                                }
155                                stack.addSlice("View" + i, ip);
156                        }
157                        GridPainter painter = new GridPainter(ip);
158                        painter.drawSquares(imagePoints[i]);
159                }
160        }
161        
162}
